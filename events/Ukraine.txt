	id = ukraine.1
	title = ukraine.1.t
	desc = ukraine.1.d
	picture = GFX_report_event_februrary_revolution

		tag = RUS

	}

		days = 7
		modifier = {
			factor = 0.75

			}
		}
	}
	
	fire_only_once = yes
	
	option = {
		name = ukraine.1.a
		set_country_flag = UKR_central_rada
	}
}
