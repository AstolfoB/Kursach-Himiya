	id = treaty_of_brest.3
	title = treaty_of_brest.3.t
	desc = treaty_of_brest.3.d
	picture = GFX_report_event_februrary_revolution

	option = {
		name = treaty_of_brest.2.a
		clr_global_flag = brest_litovsk_negotiations
		clr_global_flag = brest_litovsk_negotiations_GER_UKR_peace
		clr_global_flag = brest_litovsk_GER_UKR_negotiations
		clr_global_flag = GER_soft_peace_brest_tech_flag
		every_country = {
			limit = {
				OR = {
					has_idea = brest_negotiation_truce_Russia_and_allies
					has_idea = brest_negotiation_truce_Central_powers
				}
			}
			remove_ideas = {
				brest_negotiation_truce_Russia_and_allies
				brest_negotiation_truce_Central_powers
			}
		}
		every_country = {
			limit = { OR = { is_in_faction_with = GER is_in_faction_with = AUH AND = { is_in_faction_with = TUR NOT = { is_in_faction_with = ENG } } AND = { is_in_faction_with = BUL NOT = { is_in_faction_with = ENG } } } OR = { is_faction_leader = yes tag = AUH } }
			white_peace = RUS
			white_peace = UKR
			white_peace = USR
			white_peace = FIN
		}
		if = { limit = { owns_state = 353 } TUR = { transfer_state = 353 } }
		if = { limit = { owns_state = 354 } TUR = { transfer_state = 354 } }
		RUS = {
			add_stability = 0.2
			add_political_power = -100
			set_country_flag = RUS_brest_peace_white
			set_country_flag = RUS_brest_peace_fired
		}
		RUS = { country_event = { id = rcw.109 days = 30 } }
		add_to_variable = {
			var = global.red_alert
			value = 5
		}
	}
}
