	id = germany.20
	title = germany.20.t
	desc = germany.20.desc

	trigger = {
		original_tag = GER
		has_war_with = RUS
		any_country = {
			limit = {
				OR = {
					tag = THIS
					is_in_faction_with = THIS
				}
			}
			controls_state = 195
		}
		any_country = {
			limit = {
				OR = {
					tag = THIS
					is_in_faction_with = THIS
				}
			}
			controls_state = 219
		}
		has_global_flag = WWI_is_now
		NOT = { has_global_flag = RUS_has_capitulated_WWI }
	}
	
	option = {
		name = germany.9.a
		set_global_flag = RUS_has_capitulated_WWI
		every_country = {
			limit = { is_puppet_of = RUS has_war_with = RUS }
			white_peace = THIS
			leave_faction = yes
			RUS = {
				end_puppet = PREV
			}
		}
		RUS = {
			white_peace = THIS
			leave_faction = yes
		}
	}
}
