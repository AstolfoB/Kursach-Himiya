	}

		days = 2
	}

		hidden_effect = { remove_ideas = fascist_revolutionaries }

		name = political.21.a

			factor = 0
		}

			size = 0.5
		}
	}

		name = political.21.b

			limit = {
				has_government = democratic

				ruling_party = fascism

			}

				has_government = communism
			}

			}
