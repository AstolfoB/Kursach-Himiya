add_namespace = poland

### Независимость Польши после поражения Германии в ПМВ
country_event = {
	id = poland.1
	title = poland.1.t
	desc = poland.1.d
	picture = GFX_report_event_greece_civil_war_soldiers
	
	fire_only_once = yes
	
	trigger = {
		tag = POL
		POL = { exists = yes }
		has_global_flag = WWI_GER_HAS_CAPITULATED
		NOT = {OR = {has_government = communism has_government = radical_socialism}}
	}

	mean_time_to_happen = {
		days = 0
	}
	
	option = {
		name = poland.1.a
		set_politics = { ruling_party = despotism elections_allowed = no }
		add_popularity = { ideology = authoritarism popularity = -0.7 }
		leave_faction = yes
		if = {
		limit = {has_country_leader = { name = "Hans Beseler" }}
		retire_country_leader = yes
		}
		every_country = {
			limit = { has_war_with = POL }
			white_peace = POL
		}
		load_oob = POL_1917
		every_country = {
		limit = {POL = {is_subject_of = ROOT}}
		end_puppet = POL
		}
	}
}
